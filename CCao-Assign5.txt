USE `united_helpers`;
#1
SELECT quantityOnHand FROM item WHERE itemDescription = 'bottle of antibiotics';
#2
SELECT volunteerName FROM volunteer WHERE volunteerTelephone IS NOT NULL AND volunteerTelephone NOT LIKE '2%' AND volunteerName NOT LIKE '%Jones';
#3
SELECT v.volunteerName FROM volunteer v JOIN assignment a ON v.volunteerId = a.volunteerId JOIN task t ON a.taskCode = t.taskCode JOIN task_type tt ON t.taskTypeId = tt.taskTypeId WHERE tt.taskTypeName = 'transporting';
#4
SELECT t.taskDescription FROM task t LEFT JOIN assignment a ON t.taskCode = a.taskCode WHERE a.volunteerId IS NULL;
#5
SELECT DISTINCT pt.packageTypeName FROM package_contents pc JOIN item i ON pc.itemId = i.itemId JOIN package p ON pc.packageId = p.packageId JOIN package_type pt ON p.packageTypeId = pt.packageTypeId WHERE i.itemDescription LIKE '%bottle%';
#6
SELECT itemDescription FROM item i LEFT JOIN package_contents pc ON i.itemId = pc.itemId WHERE pc.packageId IS NULL;
#7
SELECT t.taskDescription FROM task t JOIN assignment a ON t.taskCode = a.taskCode JOIN volunteer v ON a.volunteerId = v.volunteerId WHERE v.volunteerAddress LIKE '%NJ%';
#8
SELECT v.volunteerName FROM volunteer v JOIN assignment a ON v.volunteerId = a.volunteerId WHERE a.startDateTime BETWEEN '2021-01-01' AND '2021-06-30';
#9
SELECT DISTINCT v.volunteerName FROM volunteer v JOIN assignment a ON v.volunteerId = a.volunteerId JOIN task t ON a.taskCode = t.taskCode JOIN package p ON p.taskCode = t.taskCode JOIN package_contents pc ON pc.packageId = p.packageId JOIN item i ON pc.itemId = i.itemId WHERE i.itemDescription = 'can of spam';
#10
SELECT i.itemDescription FROM item i JOIN package_contents pc ON i.itemId = pc.itemId WHERE pc.itemQuantity * i.itemValue = 100;
#11
SELECT ts.taskStatusName, COUNT(DISTINCT a.volunteerId) numVolunteers FROM task t JOIN task_status ts ON t.taskStatusId = ts.taskStatusId JOIN assignment a ON t.taskCode = a.taskCode GROUP BY t.taskStatusId ORDER BY numVolunteers DESC;
#12
SELECT p.taskCode, SUM(p.packageWeight) AS totalWeight FROM package p GROUP BY p.taskCode ORDER BY totalWeight DESC LIMIT 1;
#13
SELECT COUNT(*) AS numNonPackingTasks FROM task t JOIN task_type tt ON t.taskTypeId = tt.taskTypeId;
#14
SELECT i.itemDescription FROM item i JOIN package_contents pc ON i.itemId = pc.itemId JOIN package p ON pc.packageId = p.packageId JOIN task t ON p.taskCode = t.taskCode JOIN assignment a ON a.taskCode = t.taskCode GROUP BY i.itemDescription HAVING COUNT(DISTINCT a.volunteerId) < 3;
#15
SELECT pc.packageId, SUM(pc.itemQuantity * i.itemValue) AS packageValue FROM package_contents pc JOIN item i ON pc.itemId = i.itemId GROUP BY pc.packageId HAVING packageValue > 100 ORDER BY packageValue ASC;
